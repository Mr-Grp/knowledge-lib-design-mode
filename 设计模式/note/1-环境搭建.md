# 环境搭建
参考 webpack 第 1 2 章
webpack 要全局安装

```
npm i -D webpack webpack-cli babel-loader @babel/core @babel/preset-env @babel/polyfill html-webpack-plugin clean-webpack-plugin dev-server core-js@3
```


## webpack.config.js
```js
const path = require('path');
const HtmlWebpackPlugin = require('html-webpack-plugin');
const { CleanWebpackPlugin } = require('clean-webpack-plugin');

module.exports = {
  mode: 'development',
  devtool: 'cheap-module-eval-source-map',
  entry: {
  main: './src/index.js'
  },
  devServer: {
    contentBase: './dist',
    open: true,
    port: 9000,
    hot: true,
    hotOnly: true
  },
  module: {
    rules: [
      { 
        test: /\.js$/, 
        exclude: /node_modules/, 
        loader: 'babel-loader',
      }, {
        test: /\.css$/,
        use: [
          {
            loader: 'style-loader',
            options: {
              importLoaders: 2,
              modules: true
            }
          },
          'css-loader',
          'postcss-loader'
        ]
      }]
  },
  plugins: [
    new HtmlWebpackPlugin({
      template: 'src/index.html'
    }), 
    new CleanWebpackPlugin(),
  ],
  output: {
    filename: '[name].js',
    path: path.resolve(__dirname, 'dist')
  }
}
```
## .babelrc 文件
```js
{
  "presets": [
    [
      "@babel/preset-env",
      {
        "useBuiltIns": "usage",
        "corejs": 3
      }
    ]
  ]
}
```
